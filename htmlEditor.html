<!DOCTYPE html>

<html>

<head>

    <title>Countdown</title>

    <script>

        function updateCounter() {

            var targetDate = new Date("February 11, 2023").getTime(); // Das Ziel-Datum

            var today = new Date().getTime(); // Das aktuelle Datum


            var timeDifference = today - targetDate;

            var days = Math.floor(timeDifference / (1000 * 60 * 60 * 24)); // Berechnung der vergangenen Tage


            var counterElement = document.getElementById("counter");

            counterElement.innerHTML = "Wir sind seit " + days + " Tage zusammen.";

        }

        

        window.onload = updateCounter;

    </script>

</head>

<body>

    <h1>Countdown</h1>

    <p id="counter"></p>

</body>

</html>
